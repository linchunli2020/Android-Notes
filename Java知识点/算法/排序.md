    private static int[] testBubbleSort(int[] array) {
            int temp;
            /**
             * 外层控制循环多少趟
             * 内层控制每一趟循环次数
             */
            //需要length-1次循环比较
            for (int i = 0; i < array.length - 1; i++) {
                //内层：每次循环需要两两比较的次数，每次比较后，都会将当前最大的数放到最后位置，所以每次比较次数递减一次
                for (int j = 0; j < array.length - i - 1; j++) {
                    if (array[j] > array[j + 1]) {
                        temp = array[j];
                        array[j] = array[j + 1];
                        array[j + 1] = temp;
                    }
                }
            }
            return array;
   }
